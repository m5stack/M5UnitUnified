138inline GpioPinPair gpioPins(const GpioRole role = GpioRole::Both)
139{
140 int rx = (role == GpioRole::OutOnly) ? -1 : M5.getPin(m5::pin_name_t::port_b_in);
141 int tx = (role == GpioRole::InOnly) ? -1 : M5.getPin(m5::pin_name_t::port_b_out);
142 const bool need_rx_fb = (role != GpioRole::OutOnly && rx < 0);
143 const bool need_tx_fb = (role != GpioRole::InOnly && tx < 0);
144 bool fb = false;
145 if (need_rx_fb || need_tx_fb) {
146 fb = true;
147 rx = (role == GpioRole::OutOnly) ? -1 : M5.getPin(m5::pin_name_t::port_a_pin1);
148 tx = (role == GpioRole::InOnly) ? -1 : M5.getPin(m5::pin_name_t::port_a_pin2);
149 }
150 return {static_cast<int8_t>(rx), static_cast<int8_t>(tx), fb};
151}

298inline bool addI2C(UnitUnified& units, Component& unit, const uint32_t clock = 100000,
299 const NessoPort nesso = NessoPort::PortB)
300{
301 const auto p = i2cPins(nesso);
302 M5_LIB_LOGI("wiring: addI2C board=0x%02x nesso=%d backend=%d sda=%d scl=%d", (int)M5.getBoard(), (int)nesso,
303 (int)p.backend, (int)p.sda, (int)p.scl);
304 switch (p.backend) {
305 case I2CPins::Backend::SoftwareI2C:
306 return i2cSoftware(units, unit, p.sda, p.scl);
307 case I2CPins::Backend::Wire:
308 return i2cWire(units, unit, Wire, p.sda, p.scl, clock);
309 case I2CPins::Backend::ExI2C:
310 return i2cClass(units, unit, M5.Ex_I2C);
311 }
312 return false;
313}

353inline bool addUART(UnitUnified& units, Component& unit, const uint32_t baud = 115200, const uint32_t config = 0)
354{
355 HardwareSerial& serial = defaultUartSerial();
356 const uint32_t cfg = (config == 0) ? static_cast<uint32_t>(SERIAL_8N1) : config;
357 const auto p = uartPins();
358 if (p.fallback_a) {
359 const auto b = M5.getBoard();
360 if (b == m5::board_t::board_M5NanoC6 || b == m5::board_t::board_M5NanoH2) {
361 // M5.begin() calls Ex_I2C.setPort() but NOT begin(). m5gfx::i2c::release() is gated
362 // by the internal `initialized` flag, so release() alone is a no-op. Workaround:
363 // begin() flips initialized=true, then release() actually detaches + restores pins.
364 M5.Ex_I2C.begin();
365 M5.Ex_I2C.release();
366 }
367 }
368 M5_LIB_LOGI("wiring: UART rx=%d tx=%d baud=%lu config=0x%lx fallback_a=%d", (int)p.rx, (int)p.tx,
369 (unsigned long)baud, (unsigned long)cfg, (int)p.fallback_a);
370 serial.end();
371 serial.begin(baud, cfg, p.rx, p.tx);
372 return units.add(unit, serial);
373}

389inline bool addSPI(UnitUnified& units, Component& unit, const uint32_t clock_hz, const uint8_t mode = 0,
390 const uint8_t bit_order = 0)
391{
392 SPISettings settings{clock_hz, static_cast<uint8_t>((bit_order == 0) ? MSBFIRST : LSBFIRST), mode};
393 const auto p = spiPins();
394 M5_LIB_LOGI("wiring: addSPI sclk=%d miso=%d mosi=%d clock=%lu mode=%u bit_order=%u", (int)p.sclk, (int)p.miso,
395 (int)p.mosi, (unsigned long)clock_hz, (unsigned)mode, (unsigned)bit_order);
396 if (!SPI.bus()) {
397 SPI.begin(p.sclk, p.miso, p.mosi);
398 }
399 return spiBus(units, unit, SPI, settings);
400}

409inline bool addHatI2C(UnitUnified& units, Component& unit, const uint32_t clock = 400000)
410{
411 const auto p = hatI2CPins();
412 if (p.sda < 0 || p.scl < 0) {
413 M5_LIB_LOGE("wiring: Hat I2C unsupported board=0x%02x", (int)M5.getBoard());
414 return false;
415 }
416#if SOC_I2C_NUM > 1
417 TwoWire& wire = p.useWire1 ? Wire1 : Wire;
418#else
419 // SOC_I2C_NUM == 1 (ESP32-C3/C6/H2): Arduino-ESP32 declares Wire only; Wire1 is absent.
420 if (p.useWire1) {
421 M5_LIB_LOGE("wiring: addHatI2C NessoN1 Hat needs Wire1, but SOC_I2C_NUM==1");
422 return false;
423 }
424 TwoWire& wire = Wire;
425#endif
426 M5_LIB_LOGI("wiring: addHatI2C board=0x%02x sda=%d scl=%d clock=%lu wire=%s", (int)M5.getBoard(), (int)p.sda,
427 (int)p.scl, (unsigned long)clock, p.useWire1 ? "Wire1" : "Wire");
428 return i2cWire(units, unit, wire, p.sda, p.scl, clock);
429}

435inline bool addHatGPIO(UnitUnified& units, Component& unit, const GpioRole role = GpioRole::Both)
436{
437 auto p = hatGPIOPins();
438 if (p.rx < 0 || p.tx < 0) {
439 M5_LIB_LOGE("wiring: Hat GPIO unsupported board=0x%02x", (int)M5.getBoard());
440 return false;
441 }
442 if (role == GpioRole::OutOnly) p.rx = -1;
443 if (role == GpioRole::InOnly) p.tx = -1;
444 M5_LIB_LOGI("wiring: addHatGPIO board=0x%02x rx=%d tx=%d role=%d", (int)M5.getBoard(), (int)p.rx, (int)p.tx,
445 (int)role);
446 return units.add(unit, p.rx, p.tx);
