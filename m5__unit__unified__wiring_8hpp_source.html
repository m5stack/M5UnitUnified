101inline I2CPins i2cPins(const NessoPort nesso = NessoPort::PortB)
102{
103 const auto board = M5.getBoard();
104 if (board == m5::board_t::board_ArduinoNessoN1) {
105 if (nesso == NessoPort::PortB) {
106 return {static_cast<int8_t>(M5.getPin(m5::pin_name_t::port_b_out)),
107 static_cast<int8_t>(M5.getPin(m5::pin_name_t::port_b_in)), I2CPins::Backend::SoftwareI2C};

114 static_cast<int8_t>(M5.getPin(m5::pin_name_t::ex_i2c_scl)), I2CPins::Backend::ExI2C};
115 }
116 return {static_cast<int8_t>(M5.getPin(m5::pin_name_t::port_a_sda)),
117 static_cast<int8_t>(M5.getPin(m5::pin_name_t::port_a_scl)), I2CPins::Backend::Wire};

127inline GpioPinPair gpioPins(const GpioRole role = GpioRole::Both)
128{
129 int rx = (role == GpioRole::OutOnly) ? -1 : M5.getPin(m5::pin_name_t::port_b_in);
130 int tx = (role == GpioRole::InOnly) ? -1 : M5.getPin(m5::pin_name_t::port_b_out);
131 const bool need_rx_fb = (role != GpioRole::OutOnly && rx < 0);
132 const bool need_tx_fb = (role != GpioRole::InOnly && tx < 0);
133 bool fb = false;
134 if (need_rx_fb || need_tx_fb) {
135 fb = true;
136 rx = (role == GpioRole::OutOnly) ? -1 : M5.getPin(m5::pin_name_t::port_a_pin1);
137 tx = (role == GpioRole::InOnly) ? -1 : M5.getPin(m5::pin_name_t::port_a_pin2);
138 }
139 return {static_cast<int8_t>(rx), static_cast<int8_t>(tx), fb};
140}

287inline bool addI2C(UnitUnified& units, Component& unit, const uint32_t clock = 100000,
288 const NessoPort nesso = NessoPort::PortB)
289{
290 const auto p = i2cPins(nesso);
291 M5_LIB_LOGI("wiring: addI2C board=0x%02x nesso=%d backend=%d sda=%d scl=%d", (int)M5.getBoard(), (int)nesso,
292 (int)p.backend, (int)p.sda, (int)p.scl);
293 switch (p.backend) {
294 case I2CPins::Backend::SoftwareI2C:
295 return i2cSoftware(units, unit, p.sda, p.scl);
296 case I2CPins::Backend::Wire:
297 return i2cWire(units, unit, Wire, p.sda, p.scl, clock);
298 case I2CPins::Backend::ExI2C:
299 return i2cClass(units, unit, M5.Ex_I2C);
300 }
301 return false;
302}

339inline bool addUART(UnitUnified& units, Component& unit, HardwareSerial& serial, const uint32_t baud = 115200,
340 const uint32_t config = SERIAL_8N1)
341{
342 const auto p = uartPins();
343 if (p.fallback_a) {
344 const auto b = M5.getBoard();
345 if (b == m5::board_t::board_M5NanoC6 || b == m5::board_t::board_M5NanoH2) {
346 // M5.begin() calls Ex_I2C.setPort() but NOT begin(). m5gfx::i2c::release() is gated
347 // by the internal `initialized` flag, so release() alone is a no-op. Workaround:
348 // begin() flips initialized=true, then release() actually detaches + restores pins.
349 M5.Ex_I2C.begin();
350 M5.Ex_I2C.release();
351 }
352 }
353 M5_LIB_LOGI("wiring: UART rx=%d tx=%d baud=%lu fallback_a=%d", (int)p.rx, (int)p.tx, (unsigned long)baud,
354 (int)p.fallback_a);
355 serial.end();
356 serial.begin(baud, config, p.rx, p.tx);
357 return units.add(unit, serial);
358}

388inline bool addHatI2C(UnitUnified& units, Component& unit, const uint32_t clock = 400000)
389{
390 const auto p = hatI2CPins();
391 if (p.sda < 0 || p.scl < 0) {
392 M5_LIB_LOGE("wiring: Hat I2C unsupported board=0x%02x", (int)M5.getBoard());
393 return false;
394 }
395#if SOC_I2C_NUM > 1
396 TwoWire& wire = p.useWire1 ? Wire1 : Wire;
397#else
398 // SOC_I2C_NUM == 1 (ESP32-C3/C6/H2): Arduino-ESP32 declares Wire only; Wire1 is absent.
399 if (p.useWire1) {
400 M5_LIB_LOGE("wiring: addHatI2C NessoN1 Hat needs Wire1, but SOC_I2C_NUM==1");
401 return false;
402 }
403 TwoWire& wire = Wire;
404#endif
405 M5_LIB_LOGI("wiring: addHatI2C board=0x%02x sda=%d scl=%d clock=%lu wire=%s", (int)M5.getBoard(), (int)p.sda,
406 (int)p.scl, (unsigned long)clock, p.useWire1 ? "Wire1" : "Wire");
407 return i2cWire(units, unit, wire, p.sda, p.scl, clock);
408}

414inline bool addHatGPIO(UnitUnified& units, Component& unit, const GpioRole role = GpioRole::Both)

424 (int)role);
