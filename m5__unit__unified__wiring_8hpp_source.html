143inline I2CPins i2cPins(const NessoPort nesso = NessoPort::PortB)
144{
145 const auto board = M5.getBoard();
146 if (board == m5::board_t::board_ArduinoNessoN1) {
147 if (nesso == NessoPort::PortB) {
148 return {static_cast<int8_t>(M5.getPin(m5::pin_name_t::port_b_out)),
149 static_cast<int8_t>(M5.getPin(m5::pin_name_t::port_b_in)), I2CPins::Backend::SoftwareI2C};
150 }
151 return {static_cast<int8_t>(M5.getPin(m5::pin_name_t::port_a_sda)),
152 static_cast<int8_t>(M5.getPin(m5::pin_name_t::port_a_scl)), I2CPins::Backend::Wire};
153 }
154 if (board == m5::board_t::board_M5NanoC6 || board == m5::board_t::board_M5NanoH2) {
155 return {static_cast<int8_t>(M5.getPin(m5::pin_name_t::ex_i2c_sda)),
156 static_cast<int8_t>(M5.getPin(m5::pin_name_t::ex_i2c_scl)), I2CPins::Backend::ExI2C};
157 }
158 if (board == m5::board_t::board_M5Stack) {
159 // Core (Basic/Gray/Go/Fire): In_I2C and Ex_I2C are the same bus (I2C_NUM_0 / GPIO21,22).
160 // M5.begin() already installs that bus via In_I2C, so opening a new Wire / native master bus
161 // on the same port would collide. Borrow M5.Ex_I2C instead (ExI2C backend -> i2cClass).
162 return {static_cast<int8_t>(M5.getPin(m5::pin_name_t::ex_i2c_sda)),
163 static_cast<int8_t>(M5.getPin(m5::pin_name_t::ex_i2c_scl)), I2CPins::Backend::ExI2C};
164 }
165 return {static_cast<int8_t>(M5.getPin(m5::pin_name_t::port_a_sda)),
166 static_cast<int8_t>(M5.getPin(m5::pin_name_t::port_a_scl)), I2CPins::Backend::Wire};
167}

176inline GpioPinPair gpioPins(const GpioRole role = GpioRole::Both)
177{
178 int rx = (role == GpioRole::OutOnly) ? -1 : M5.getPin(m5::pin_name_t::port_b_in);
179 int tx = (role == GpioRole::InOnly) ? -1 : M5.getPin(m5::pin_name_t::port_b_out);
180 const bool need_rx_fb = (role != GpioRole::OutOnly && rx < 0);
181 const bool need_tx_fb = (role != GpioRole::InOnly && tx < 0);
182 bool fb = false;
183 if (need_rx_fb || need_tx_fb) {
184 fb = true;
185 rx = (role == GpioRole::OutOnly) ? -1 : M5.getPin(m5::pin_name_t::port_a_pin1);
186 tx = (role == GpioRole::InOnly) ? -1 : M5.getPin(m5::pin_name_t::port_a_pin2);
187 }
188 return {static_cast<int8_t>(rx), static_cast<int8_t>(tx), fb};
189}

338{

414inline bool addUART(UnitUnified& units, Component& unit, const uint32_t baud = 115200,
415 const UartConfig config = UartConfig::Default)
416{
417 HardwareSerial& serial = defaultUartSerial();
418 const uint32_t cfg = toArduinoSerialConfig(config);
419 const auto p = uartPins();
420 if (p.fallback_a) {
421 const auto b = M5.getBoard();
422 if (b == m5::board_t::board_M5NanoC6 || b == m5::board_t::board_M5NanoH2) {
423 // M5.begin() calls Ex_I2C.setPort() but NOT begin(). m5gfx::i2c::release() is gated
424 // by the internal `initialized` flag, so release() alone is a no-op. Workaround:
425 // begin() flips initialized=true, then release() actually detaches + restores pins.
426 M5.Ex_I2C.begin();
427 M5.Ex_I2C.release();
428 }
429 }
430 M5_LIB_LOGI("wiring: UART rx=%d tx=%d baud=%lu config=0x%lx fallback_a=%d", (int)p.rx, (int)p.tx,
431 (unsigned long)baud, (unsigned long)cfg, (int)p.fallback_a);
432 serial.end();
433 serial.begin(baud, cfg, p.rx, p.tx);
434 return units.add(unit, serial);
435}

451inline bool addSPI(UnitUnified& units, Component& unit, const uint32_t clock_hz, const uint8_t mode = 0,
452 const uint8_t bit_order = 0)
453{
454 SPISettings settings{clock_hz, static_cast<uint8_t>((bit_order == 0) ? MSBFIRST : LSBFIRST), mode};
455 const auto p = spiPins();
456 M5_LIB_LOGI("wiring: addSPI sclk=%d miso=%d mosi=%d clock=%lu mode=%u bit_order=%u", (int)p.sclk, (int)p.miso,
457 (int)p.mosi, (unsigned long)clock_hz, (unsigned)mode, (unsigned)bit_order);
458 if (!SPI.bus()) {
459 SPI.begin(p.sclk, p.miso, p.mosi);
460 }
461 return spiBus(units, unit, SPI, settings);
462}

471inline bool addHatI2C(UnitUnified& units, Component& unit, const uint32_t clock = 400000)
472{
473 const auto p = hatI2CPins();
474 if (p.sda < 0 || p.scl < 0) {
475 M5_LIB_LOGE("wiring: Hat I2C unsupported board=0x%02x", (int)M5.getBoard());
476 return false;
477 }
478#if SOC_I2C_NUM > 1
479 TwoWire& wire = p.useWire1 ? Wire1 : Wire;
480#else
481 // SOC_I2C_NUM == 1 (ESP32-C3/C6/H2): Arduino-ESP32 declares Wire only; Wire1 is absent.
482 if (p.useWire1) {
483 M5_LIB_LOGE("wiring: addHatI2C NessoN1 Hat needs Wire1, but SOC_I2C_NUM==1");
484 return false;
485 }
486 TwoWire& wire = Wire;
487#endif
488 M5_LIB_LOGI("wiring: addHatI2C board=0x%02x sda=%d scl=%d clock=%lu wire=%s", (int)M5.getBoard(), (int)p.sda,
489 (int)p.scl, (unsigned long)clock, p.useWire1 ? "Wire1" : "Wire");
490 return i2cWire(units, unit, wire, p.sda, p.scl, clock);
491}

610#if __has_include(<driver/i2c_master.h>)
612inline i2c_master_bus_handle_t ensureI2CBus(const i2c_port_t port, const gpio_num_t sda, const gpio_num_t scl,
613 const uint32_t /*clock - not used to key, single bus per pins*/)
614{
615 const uint32_t key = (static_cast<uint32_t>(static_cast<uint8_t>(port)) << 24) |
616 (static_cast<uint32_t>(static_cast<uint8_t>(sda)) << 16) |
617 (static_cast<uint32_t>(static_cast<uint8_t>(scl)) << 8);
618 static I2CCacheEntry cache[kI2CBusCacheSize]{};
619 static size_t count{};
620 for (size_t i = 0; i < count; ++i) {
621 if (cache[i].key == key) return cache[i].handle;
622 }
623 if (count >= kI2CBusCacheSize) {
624 M5_LIB_LOGE("wiring: I2C bus cache full (max %zu)", kI2CBusCacheSize);
625 return nullptr;
626 }
627 i2c_master_bus_config_t cfg{};
628 cfg.i2c_port = port;
629 cfg.sda_io_num = sda;
630 cfg.scl_io_num = scl;
631#if SOC_LP_I2C_SUPPORTED
632 if (port == LP_I2C_NUM_0) {
633 // LP I2C uses RTC_FAST clock (union with clk_source, mutually exclusive).
634 cfg.lp_source_clk = LP_I2C_SCLK_DEFAULT;
635 } else
636#endif
637 {
638 cfg.clk_source = I2C_CLK_SRC_DEFAULT;
639 }
640 cfg.glitch_ignore_cnt = 7;
641 cfg.flags.enable_internal_pullup = true;
642 if (i2c_new_master_bus(&cfg, &cache[count].handle) != ESP_OK) {
643 M5_LIB_LOGE("wiring: i2c_new_master_bus failed port=%d sda=%d scl=%d", (int)port, (int)sda, (int)scl);
644 return nullptr;
645 }
646 cache[count].key = key;
647 return cache[count++].handle;
648}
649#else // legacy driver/i2c.h (IDF 5.0 / 5.1)
651inline bool ensureI2CLegacyDriver(const i2c_port_t port, const gpio_num_t sda, const gpio_num_t scl,
652 const uint32_t clock)
653{
654 const uint32_t key = (static_cast<uint32_t>(static_cast<uint8_t>(port)) << 24) |
655 (static_cast<uint32_t>(static_cast<uint8_t>(sda)) << 16) |
656 (static_cast<uint32_t>(static_cast<uint8_t>(scl)) << 8);
657 static uint32_t cache[kI2CBusCacheSize]{};
658 static size_t count{};
659 for (size_t i = 0; i < count; ++i) {
660 if (cache[i] == key) return true;
661 }
662 if (count >= kI2CBusCacheSize) {
663 M5_LIB_LOGE("wiring: I2C bus cache full (max %zu)", kI2CBusCacheSize);
664 return false;
665 }
666 i2c_config_t conf{};
667 conf.mode = I2C_MODE_MASTER;
668 conf.sda_io_num = sda;
669 conf.scl_io_num = scl;
670 conf.sda_pullup_en = true;
671 conf.scl_pullup_en = true;
672 conf.master.clk_speed = clock;
673 if (i2c_param_config(port, &conf) != ESP_OK || i2c_driver_install(port, I2C_MODE_MASTER, 0, 0, 0) != ESP_OK) {
674 M5_LIB_LOGE("wiring: legacy i2c driver install failed port=%d sda=%d scl=%d", (int)port, (int)sda, (int)scl);
675 return false;
676 }
677 cache[count++] = key;
678 return true;
679}

683inline uart_port_t ensureUARTPort(const uart_port_t port, const gpio_num_t rx, const gpio_num_t tx, const uint32_t baud,
684 const UartConfig config)
685{
686 const uint32_t key = (static_cast<uint32_t>(static_cast<uint8_t>(port)) << 24) |
687 (static_cast<uint32_t>(static_cast<uint8_t>(rx)) << 16) |
688 (static_cast<uint32_t>(static_cast<uint8_t>(tx)) << 8);
689 static UARTCacheEntry cache[kUARTPortCacheSize]{};
690 static size_t count{};
691 for (size_t i = 0; i < count; ++i) {
692 if (cache[i].key == key) return cache[i].port;
693 }
694 if (count >= kUARTPortCacheSize) {
695 M5_LIB_LOGE("wiring: UART port cache full (max %zu)", kUARTPortCacheSize);
696 return UART_NUM_MAX;
697 }
698 if (uart_is_driver_installed(port)) {
699 cache[count] = {key, port};
700 return cache[count++].port;
701 }
702 const auto p = toIdfUartParams(config);
703 uart_config_t uc{};
704 uc.baud_rate = static_cast<int>(baud);
705 uc.data_bits = p.word_length;
706 uc.parity = p.parity;
707 uc.stop_bits = p.stop_bits;
708 uc.flow_ctrl = UART_HW_FLOWCTRL_DISABLE;
709 uc.source_clk = UART_SCLK_DEFAULT;
710 if (uart_param_config(port, &uc) != ESP_OK ||
711 uart_set_pin(port, tx, rx, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE) != ESP_OK ||
712 uart_driver_install(port, 256, 0, 0, nullptr, 0) != ESP_OK) {
713 M5_LIB_LOGE("wiring: uart setup failed port=%d", (int)port);
714 return UART_NUM_MAX;
715 }
716 cache[count] = {key, port};
717 return cache[count++].port;
718}

722inline spi_device_handle_t ensureSPIDevice(const spi_host_device_t host, const gpio_num_t mosi, const gpio_num_t miso,
723 const gpio_num_t sck, const uint32_t clock_hz, const uint8_t mode,
724 const uint8_t bit_order)
725{
726 static SPIHostEntry host_cache[kSPIHostCacheSize]{};
727 static SPIDevEntry dev_cache[kSPIDevCacheSize]{};
728 static size_t host_count{};
729 static size_t dev_count{};
730
731 bool host_init = false;
732 for (size_t i = 0; i < host_count; ++i) {
733 if (host_cache[i].host == host) {
734 host_init = host_cache[i].initialized;
735 break;
736 }
737 }
738 if (!host_init) {
739 spi_bus_config_t bc{};
740 bc.mosi_io_num = mosi;
741 bc.miso_io_num = miso;
742 bc.sclk_io_num = sck;
743 bc.quadwp_io_num = -1;
744 bc.quadhd_io_num = -1;
745 bc.max_transfer_sz = 4096;
746 if (spi_bus_initialize(host, &bc, SPI_DMA_CH_AUTO) != ESP_OK) {
747 M5_LIB_LOGE("wiring: spi_bus_initialize failed host=%d", (int)host);
748 return nullptr;
749 }
750 if (host_count < kSPIHostCacheSize) host_cache[host_count++] = {host, true};
751 }
752
753 const uint32_t key = (static_cast<uint32_t>(static_cast<uint8_t>(host)) << 24) |
754 (static_cast<uint32_t>(clock_hz / 1000U) << 8) | (static_cast<uint32_t>(mode) << 4) |
755 static_cast<uint32_t>(bit_order);
756 for (size_t i = 0; i < dev_count; ++i) {
757 if (dev_cache[i].key == key) return dev_cache[i].handle;
758 }
759 if (dev_count >= kSPIDevCacheSize) {
760 M5_LIB_LOGE("wiring: SPI device cache full (max %zu)", kSPIDevCacheSize);
761 return nullptr;
762 }
763 spi_device_interface_config_t dc{};
764 dc.clock_speed_hz = static_cast<int>(clock_hz);
765 dc.mode = mode;
766 dc.spics_io_num = GPIO_NUM_NC; // manual CS control by adapter
767 dc.queue_size = 1;
768 if (bit_order == 1) dc.flags |= SPI_DEVICE_BIT_LSBFIRST;
769 if (spi_bus_add_device(host, &dc, &dev_cache[dev_count].handle) != ESP_OK) {
770 M5_LIB_LOGE("wiring: spi_bus_add_device failed host=%d", (int)host);
771 return nullptr;
772 }
773 dev_cache[dev_count].key = key;
774 return dev_cache[dev_count++].handle;
775}

827inline bool addI2C(UnitUnified& units, Component& unit, const uint32_t clock = 100000,
828 const NessoPort nesso = NessoPort::PortB)
829{
830 const auto pins = i2cPins(nesso);
831 const auto board = M5.getBoard();
832 M5_LIB_LOGI("wiring(ESP-IDF): addI2C board=0x%02x nesso=%d sda=%d scl=%d clock=%u backend=%d", (int)board,
833 (int)nesso, (int)pins.sda, (int)pins.scl, (unsigned)clock, (int)pins.backend);
834 switch (pins.backend) {
835 case I2CPins::Backend::SoftwareI2C:
836#if defined(M5_HAL_HPP)
837 return i2cSoftware(units, unit, pins.sda, pins.scl);
838#else
839 M5_LIB_LOGE("wiring: SoftwareI2C backend requires M5HAL");
840 return false;
841#endif
842 case I2CPins::Backend::Wire:
843 if (board == m5::board_t::board_ArduinoNessoN1) {
844 // NessoN1 PortA (QWIIC) -> Wire == M5.In_I2C (NessoN1 maps Wire to In_I2C).
845 // M5Unified already holds I2C_NUM_0 here, so borrow instead of installing.
846 return i2cClass(units, unit, M5.In_I2C);
847 } else {
848 // Stick / S3 etc: M5Unified does not call Wire.begin(), so install ourselves
849 // (Core is routed to ExI2C by i2cPins() and never reaches this branch)
850#if __has_include(<driver/i2c_master.h>)
851 auto bus = i2cBusHandle(I2C_NUM_0, (gpio_num_t)pins.sda, (gpio_num_t)pins.scl, clock);
852 if (!bus) return false;
853 return units.add(unit, bus);
854#else // legacy driver (IDF 5.0 / 5.1): install legacy driver, then add by port
855 if (!detail::ensureI2CLegacyDriver(I2C_NUM_0, (gpio_num_t)pins.sda, (gpio_num_t)pins.scl, clock)) {
856 return false;
857 }
858 return units.add(unit, I2C_NUM_0, (gpio_num_t)pins.sda, (gpio_num_t)pins.scl);
859#endif
860 }
861 case I2CPins::Backend::ExI2C:
862 // Core / NanoC6 / NanoH2: M5Unified manages Ex_I2C (Core shares it with In_I2C) -> borrow
863 return i2cClass(units, unit, M5.Ex_I2C);
864 }
865 return false;
866}
